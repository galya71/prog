#include <iostream>


void copy_n(const char* src, char* dest, size_t count)
{
	while (count != 0 && *src)
	{
		// копируем символ
		*dest = *src;
		// переходим к следующему символу
		++dest;
		++src;
		// уменьшаем счетчик символов
		--count;
	}


	*dest = '\0';
}
int main()
{
	setlocale(LC_ALL, "rus");
	char result[50];
	char str[50];
	gets_s(str);
	int n;
	std::cout << "Введите количество n символов, которое нужно скопировать: "; std::cin >> n;
	if (n == 0) std::cout << "Вы ввели 0 символов\n";
	if (n > 0)
	{
		copy_n(str, result, n);
		std::cout << result;
	}
}
