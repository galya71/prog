#include <iostream>
#include <ctime>

using namespace std;

void show_map(int map[3][3])
{
	for (int i = 0; i < 3; i++)
	{
		for (int j = 0; j < 3; j++)
		{
			cout << map[j][i];
		}
		cout << endl;
	}
}

bool GameProcess(int map[3][3])//проверка игрока
{
	int c = 0;//счетчик строк
	int d = 0;//счетчик столбцов
	int b = 0;//счетчик главной диагонали
	int a = 0;//счетчик второй диагонали

	for (int i = 0; i < 3; i++)
	{
		for (int j = 0; j < 3; j++)
		{
			if (map[j][i] == 1)
				d++;
			if (map[i][j] == 1)
				c++;
			if (map[j][j] == 1)
				b++;
			if (map[2 - j][j] == 1)
				a++;
		}

		if (c == 3 || d == 3 || b == 3 || a == 3)
		{
			return true;
			break;
		}

		c = 0;
		d = 0;
		b = 0;
		a = 0;
	}
	return false;
}

			return true;
			break;
		}

		a = 0;
	}

}
