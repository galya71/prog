void Searching_For_Longest_Word(char* mass, char* word, int length)
{
	int max = 0, k = 0;
	int calculations = 0;
	for (int i = 40; i < length; i++)
	{
		if (mass[i] == '\t') calculations = 0;
		if (mass[i] != '\t')
		{
			calculations++;
			if (max < calculations)
			{
				max = calculations;
				k = i;
			}
		}

	}

	int v = 0;
	k++;
	for (int i = k - max; i <= k; i++)
	{
		word[v] = mass[i];
		v++;
	}
	word[v] = '\0';

	std::cout << "Самое длинное слово: " << word << "\n";
	std::cout << "Длина" << max << "\n" << "\n";
}

void Parting(char* str, int length)
{
	for (int i = 0; i < length; i++)
	{
		if (str[i] == ' ' || str[i] == '?' || str[i] == '!' || str[i] == '.')
		{
			str[i] = '\t';
		}

	}
}
