#include <iostream>
#include <fstream>
#include <string>

using namespace std;

int main()
{
    string words[1000]; 
    string s; 
    
	ifstream f("C:\\Users\\Галя\\Desktop\\учеба ВУЗ\\языки\\lab_3\\text.txt"); 
    int cnt=0; 
    while(f >> s){ 
        words[cnt] = s; 
        cnt++; 
    }
	f.close(); 
    
    for(int i=0; i<cnt; i++){ // сортируем массив слов 
        for(int j=cnt-1; j>i; j--){ 
            if(words[j].length() < words[j-1].length()){ 
                string t = words[j]; 
                words[j] = words[j-1]; 
                words[j-1] = t;		
            }
        }
    }
    

    for(int i=0; i<cnt; i++) // проходим по отсортированному массиву слов
        cout << words[i] << endl; // выводим очередное слово

    return 0;
}
