#include <iostream>


void copy_n(const char* src, char* dest, size_t count)
{
	while (count != 0 && *src)
	{
		// копируем символ
		*dest = *src;
		// переходим к следующему символу
		++dest;
		++src;
		// уменьшаем счетчик символов
		--count;
	}


	*dest = '\0';
}
int main()
{
	char result[50];
	copy_n("Do you like programming?", result, 11);
	std::cout << result;
}
