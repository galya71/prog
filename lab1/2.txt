#include<iostream>
using namespace std;

int main()
{
	int b, a;
	cout << "Enter a: "; cin >> b;
	cout << "Enter b: "; cin >> a;
	int* arr;
	arr = new int[a - b];
	int cur = b;
	for (int i = 0; i <= a - b; i++) //Заполнение массива
		arr[i] = cur++;
	for (int i = 0; i <= a - b; i++) //Вывод массива
		cout << arr[i] << " ";
	cout << endl;
	return 0;
}
