#include"iostream"
#include "cstdlib"

using namespace std;
void check(int& n)//количество чисел в файле
{
	int value;
	string path = "C:\\мое\\лаба1\\file.txt";
	ifstream F;
	F.open(path);//открываем файл для чтения
	if (F.is_open())
	{
		F >> value;
		while (!F.eof())//проверяем, достигнут ли конец файла
		{
			F >> value;//считываем значение из файла просто чтобы !F.eof() работала правильно
			n++;
		}
		cout << "Количество чисел в файле:" << n << endl;
	}
	F.close();
}

int main()
{
	cout << "1)" << endl;
	setlocale(LC_ALL, "russian");
	int n = 1;
	check(n);
	int* a = new int[n];
	string path = "C:\\мое\\лаба1\\file.txt";
	ifstream F;
	F.open(path);
	for (int i = 0; i < n; i++)
	{
		F >> a[i]; //считываем из файла и записываем в массив 
	}
	if (!F.is_open())
	{
		cout << "Ошибка открытия файла\n";
	}
	else
	{
		cout << "Файл открыт\n";
	}
	F.close();

	//упорядочить по возрастанию
	for (int i = 0; i < n - 1; i++)
	{
		for (int j = 0; j < n - i - 1; j++)
		{
			if (a[j] > a[j + 1]) {
				// меняем элементы местами
				int temp = a[j];
				a[j] = a[j + 1];
				a[j + 1] = temp;
			}
		}
	}
	cout << "Массив из отрицательных чисел:  ";
	for (int i = 0; i < n; i++)
	{
		if (a[i] < 0)
		{
			cout << a[i] << " ";
		}
	}
	cout << endl;


	cout << "2)" << endl;
	int N, k, A, B;
	cout << "A:"; cin >> A;
	cout << "B:"; cin >> B;
	cout << "N:"; cin >> N;
	a = new int[N];
	setlocale(LC_ALL, "russian");
	cout << "Исходный массив: ";
	for (int i = 0; i < N; i++)
	{
		a[i] = A + rand() % ((B + 1) - A);
		cout << " " << a[i];
	}
	cout << endl;
	cout << "k="; cin >> k;
	int c = 0;
	int nechet = 0;
	//кол-во нечетных элементов 
	while (c < N) {
		if (a[c] % 2 == 1) {
			nechet += 1;
		}
		c++;
	}
	//увличиваем массив
	int* b = new int[N + nechet];
	int bcnt = 0;
	for (int i = 0; i < N; i++)
	{
		if (!(a[i] % 2 == 0)) // проверяем значение на нечетность
		{
			b[bcnt++] = k;  // если нечетное, то помещаем в новый массив число Х          
		}
		b[bcnt++] = a[i]; // помещаем в новый массив значение из первоначального
	}
	cout << "Новый массив: ";
	for (int i = 0; i < bcnt; i++)
		cout << b[i] << " ";
	cout << endl;
	return 0;
}
